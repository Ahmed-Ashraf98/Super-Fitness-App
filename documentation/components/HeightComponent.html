import { Component, Input } from '@angular/core';
import { CommonModule } from '@angular/common';
import { FormGroup } from '@angular/forms';

@Component({
  selector: 'app-height',
  standalone: true,
  imports: [CommonModule],
  templateUrl: './height.component.html',
  styleUrl: './height.component.scss',
})
export class HeightComponent {
  @Input() form!: FormGroup;
  @Input() heights: number[] = [];
  @Input() selectedHeightIndex!: number;
  @Input() positionClassHeight!: (i: number) => string;
  @Input() onPickHeight!: (h: number) => void;
  @Input() currentStep!: number;
  @Input() stepsLength!: number;
}

<div class="step-header mb-5">
  <div class="arc"></div>
  <div class="step-indicator">{{ currentStep }}/{{ stepsLength }}</div>
</div>

<h2 class="picker-title">What's Your Height?</h2>
<p class="picker-subtitle">This Helps Us Create Your Personalized Plan</p>
<h1 class="text-[#ff4500] m-5 text-center">CM</h1>
<div class="picker-list">
  <div
    *ngFor="let height of heights; let i = index"
    class="picker-item"
    [ngClass]="positionClassHeight(i)"
    (click)="onPickHeight(height)"
  >
    {{ height }}
    <span *ngIf="positionClassHeight(i) === 'center'" class="arrow-up"></span>
  </div>
</div>
